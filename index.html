package com.pdf.pty;


import com.google.zxing.BarcodeFormat;
import com.google.zxing.EncodeHintType;
import com.google.zxing.MultiFormatWriter;
import com.google.zxing.WriterException;
import com.google.zxing.client.j2se.MatrixToImageWriter;
import com.google.zxing.common.BitMatrix;
import com.google.zxing.qrcode.decoder.ErrorCorrectionLevel;

import java.io.File;
import java.io.IOException;
import java.nio.file.Path;
import java.util.HashMap;

public class QRCodeUtil {
    // 二维码尺寸
    private static final int QRCODE_SIZE = 500;
    // LOGO宽度
    private static final int WIDTH = 400;
    // LOGO高度
    private static final int HEIGHT = 400;

   public static void createQrcode() throws WriterException, IOException {
       //二维码中的内容，这个内容一般都需要传参过来，我这里先写死
       String contents = "http://baidu.com";
       HashMap<EncodeHintType, Object> map = new HashMap<>();
       //设置字符集
       map.put(EncodeHintType.CHARACTER_SET,"utf-8");
       //设置纠错等级L/M/Q/H,等级越高越不容易识别，等级最高为H
       map.put(EncodeHintType.ERROR_CORRECTION, ErrorCorrectionLevel.M);
       //设置边框
       map.put(EncodeHintType.MARGIN,2);
       //设置生成的图片的后缀名
       String format = "png";
       BitMatrix bitMatrix = new MultiFormatWriter().encode(contents, BarcodeFormat.QR_CODE, WIDTH, HEIGHT, map);
       //设置输出地址
       Path file=new File("D:/new.png").toPath();
       MatrixToImageWriter.writeToPath(bitMatrix,format,file);
   }
}

一般来说上面的工具类已经够用了，但是有时候我们想搞点不一样的，比如说在二维码上面添加一个logo，这样让二维码看起来不一样，于是就有了下面这个工具类

带logo

package com.pdf.pty;

import com.google.zxing.BarcodeFormat;
import com.google.zxing.EncodeHintType;
import com.google.zxing.MultiFormatWriter;
import com.google.zxing.WriterException;
import com.google.zxing.client.j2se.MatrixToImageConfig;
import com.google.zxing.client.j2se.MatrixToImageWriter;
import com.google.zxing.common.BitMatrix;
import com.itextpdf.text.pdf.qrcode.ErrorCorrectionLevel;

import javax.imageio.ImageIO;
import java.awt.*;
import java.awt.geom.RoundRectangle2D;
import java.awt.image.BufferedImage;
import java.io.File;
import java.io.IOException;
import java.util.HashMap;

public class QRCodeUtil {
    //二维码高度和宽度
    public static final int WIDTH = 400;
    private static final int HEIGHT = 400;

    public static void createCode() throws WriterException, IOException {
        HashMap<EncodeHintType, Object> map = new HashMap<>();
        //二维码内容
        String contents = "http://baidu.com";

        //设置字符集
        map.put(EncodeHintType.CHARACTER_SET, "utf-8");
        //设置纠错等级
        map.put(EncodeHintType.ERROR_CORRECTION, ErrorCorrectionLevel.M);
        //设置边框
        map.put(EncodeHintType.MARGIN, 2);

        //生成二维码对应的位矩阵对象
        BitMatrix bitMatrix = new MultiFormatWriter().encode(contents, BarcodeFormat.QR_CODE, WIDTH, HEIGHT, map);
        //设置位矩阵转图片的参数
        MatrixToImageConfig config = new MatrixToImageConfig(0xFF000001, 0xFFFFFFFF);

        // 位矩阵对象转BufferedImage对象
        BufferedImage qrcode = MatrixToImageWriter.toBufferedImage(bitMatrix, config);
        //将不带logo的输出
        ImageIO.write(qrcode, "png", new File("D://1.png"));

        BufferedImage QRcode = createQRCodeWithLogo(qrcode, new File("D:\\logo.png"));
		//将带logo的输出
        ImageIO.write(QRcode, "png", new File("D://code.png"));
        System.out.println("输出成功");
    }


	//增加logo
    public static BufferedImage createQRCodeWithLogo(BufferedImage bufferedImage, File logofile) throws IOException {

        //创建绘图对象
        Graphics2D g2 = bufferedImage.createGraphics();


//设置logo的宽高,这里除4是为了让图片比二维码小，这里有个问题，可能是因为我的高和宽都设置为400，我一开始除3，发现生成的带logo的二维码无法识别，我猜想是因为400/3是一个浮点数，导致图片不全，当我除4、 5的时候就可以了。

        int height = bufferedImage.getHeight()/5;
        int width = bufferedImage.getWidth()/5;

        //设置logo在二维码图片上的起始坐标
        int x = (bufferedImage.getWidth()-width)/2;
        int y = (bufferedImage.getHeight()-height)/2;

        //读取logo文件
        BufferedImage logo = ImageIO.read(logofile);

        //开始绘制图像
        g2.drawImage(logo,x,y,width,height,null);

        BasicStroke stroke = new BasicStroke(5,BasicStroke.CAP_ROUND,BasicStroke.JOIN_ROUND);
        g2.setStroke(stroke);// 设置笔画对象
        //指定弧度的圆角矩形
        RoundRectangle2D.Float round = new RoundRectangle2D.Float(x,y,width,height,20,20);
        g2.setColor(Color.white); //指定圆弧边框的颜色
        g2.draw(round);

        g2.dispose();
        bufferedImage.flush();
        return bufferedImage;
    }
}

经测试发现，如果加入了圆弧边框，logo图片设置为二维码大小的1/4也是扫不出来的，1/5可以扫出来

注解

元注解

元注解的作用就是负责注解其他注解，java定义了4个标准的meta-annotation类型，他们被用来提供对其他annotation类型做说明。

  • @Target：用于描述注解的使用范围（即：被描述的注解可以用在什么地方）
  • @Retention：表示需要在什么级别保存该注解信息，用户描述注解的生命周期（source<class<runtime)
  • @Document：说明该注解将包含在javadoc中
  • @Inherited：说明子类可以继承父类中的该注解
